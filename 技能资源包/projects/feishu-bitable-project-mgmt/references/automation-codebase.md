```javascript
// 规则1：新项目创建工作流
{
  "name": "新项目创建",
  "trigger": "记录创建时",
  "actions": [

    {"type": "set_field", "field": "创建时间", "value": "NOW()"},

  ]

// 规则2：状态变更工作流
{
  "name": "状态变更",
  "trigger": "项目阶段变更时",
  "conditions": [
    {"field": "新阶段", "equals": "验收"}
  ],
  "actions": [
    {"type": "log_history"},
    {"type": "mention", "target": "质量负责人"}
  ]
}

// 规则3：逾期提醒工作流
{
  "name": "逾期提醒",
  "trigger": "每日检查",
  "conditions": [
    {"field": "计划结束日期", "less_than": "TODAY()"},
    {"field": "状态", "not_equals": "已完成"}

  "actions": [
    {"type": "update_field", "field": "状态", "value": "已延期"},

  ]
}
```

## 四、公式函数库

### 4.1 项目进度监控公式

```excel
// 项目完成率
项目完成率 = COUNTIF({任务状态}, "已完成") / COUNTA({任务ID})

// 里程碑达成预警
里程碑状态 = IF(
  AND({计划完成日期}<TODAY(), {状态}<>"已完成"),
  "已延期",
  IF({完成度}=100%, "已完成", "进行中")
)

// 风险等级计算
风险等级 = IF(
  {延期天数}>30, "🔴 高风险",
  IF({延期天数}>7, "🟡 中风险", "🟢 低风险")
)
```

### 4.2 EVM挣值分析公式

```excel
// 挣值(EV)
EV = 完成工作量 × 预算单价

// 计划价值(PV)
PV = 计划工作量 × 预算单价

// 实际成本(AC)
AC = 实际花费总额

// 进度绩效指数(SPI)
SPI = EV / PV

// 成本绩效指数(CPI)
CPI = EV / AC

// 预计完成成本(EAC)
EAC = BAC / CPI

// 完工尚需估算(ETC)
ETC = EAC - AC

// 成本偏差(CV)
CV = EV - AC

// 进度偏差(SV)
SV = EV - PV
```

### 4.3 成本预警公式

```excel
// 成本偏差率
成本偏差率 = ({实际成本} - {预算基线}) / {预算基线}

// 成本预警等级
成本预警等级 = SWITCH(TRUE(),
  {成本偏差率} > 0.2, "🔴 严重超支",
  {成本偏差率} > 0.15, "🟠 重大超支",
  {成本偏差率} > 0.1, "🟡 一般超支",
  {成本偏差率} > 0.05, "🔵 轻微超支",
  "✅ 正常"
)
```

### 4.4 资源调度公式

```excel
// 资源冲突指数
资源冲突指数 = (COUNTIFS(
  {资源分配表::负责人}, @当前资源,
  {资源分配表::计划开始日期}, ">="&TODAY(),
  {资源分配表::计划开始日期}, "<"&DATEADD(TODAY(), 7, "day")
) / {资源最大负荷}) * 100

// 冲突等级
冲突等级 = IF({资源冲突指数} > 120, "🔴 严重冲突",
           IF({资源冲突指数} > 100, "🟠 高度冲突",
           IF({资源冲突指数} > 80, "🟡 中度冲突", "✅ 正常")))

// 调度优先级
调度优先级 = (
  {项目重要性} * 0.3 + 
  {任务紧急度} * 0.25 + 
  {资源技能匹配度} * 0.2 + 
  {成本影响度} * 0.15 + 
  {客户满意度影响} * 0.1
)
```

### 4.5 质量健康度公式

```excel
// 质量健康度
质量健康度 = (
  {需求覆盖率} * 0.2 + 
  {设计符合度} * 0.15 + 
  {代码质量分} * 0.25 + 
  {测试通过率} * 0.2 + 
  {用户满意度} * 0.2
) * 100

// 缺陷加权评估
缺陷加权 = COUNTIFS({严重度},"致命")*10 + 
            COUNTIFS({严重度},"严重")*5 + 
            COUNTIFS({严重度},"一般")*2 + 
            COUNTIFS({严重度},"轻微")*1

// 缺陷收敛率
缺陷收敛率 = ({上周缺陷数} - {本周缺陷数}) / {上周缺陷数}
```

### 4.6 IMR信息代谢率公式

```excel
// 信息代谢率
IMR = (信息摄入量 × 信息转化率 × 信息输出效率) / 信息熵基数

// 信息摄入量
信息摄入量 = SUM(文档数量×权重) + SUM(沟通频次×时长权重)

// 信息转化率
信息转化率 = 有效决策数 / 信息摄入总量

// IMR健康度
IMR健康度 = IF(IMR指数>85,"🟢 优秀",
           IF(IMR指数>70,"🟡 良好",
           IF(IMR指数>50,"🟠 一般","🔴 阻塞")))
```

### 4.7 决策熵减公式

```excel
// 初始熵值
初始熵值 = -SUM(各选项概率 × LOG2(各选项概率))

// 熵减幅度
熵减幅度 = 初始熵值 - 决策后熵值

// 熵减效率
熵减效率 = 熵减幅度 / 决策耗时(小时)

// 决策质量指数
决策质量指数 = (熵减幅度 × 0.5) + (执行一致性 × 0.3) + (预期偏差率 × 0.2)
```

|-----|---------|---------|
| record_id重复 | 导入数据唯一性冲突 | 使用GUID重新生成 |
